### Create Service
```
sudo tee /etc/systemd/system/empowerd.service << EOF
[Unit]
Description=empower-testnet
After=network-online.target

[Service]
User=$USER
ExecStart=$(which empowerd) start
RestartSec=3
Restart=on-failure
LimitNOFILE=65535

[Install]
WantedBy=multi-user.target
EOF
```
