### Create Service
```
sudo tee /etc/systemd/system/saod.service > /dev/null <<EOF
[Unit]
Description=saod
After=network-online.target

[Service]
User=$USER
ExecStart=$(which saod) start
Restart=on-failure
RestartSec=3
LimitNOFILE=65535

[Install]
WantedBy=multi-user.target
EOF
```
