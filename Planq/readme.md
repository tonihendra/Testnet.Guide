sudo tee /etc/systemd/system/planqd.service > /dev/null << EOF
[Unit]
Description=planq
After=network-online.target
[Service]
User=$USER
ExecStart=$(which planqd) start
Restart=always
RestartSec=3
LimitNOFILE=65535
[Install]
WantedBy=multi-user.target
EOF
